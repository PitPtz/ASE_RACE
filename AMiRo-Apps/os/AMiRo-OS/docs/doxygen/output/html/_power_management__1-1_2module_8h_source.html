73 static inline float moduleADC2V(adcsample_t sample)
74 {
75  // apply multiplications according to hard-wired voltage divider
76  return (float)sample / (float)ADC_HTR_HT * 3.3f * 5.0f;
77 }

86 static inline adcsample_t moduleV2ADC(float volts)
87 {
88  // apply divisions according to hard-wired voltage divider
89  return (adcsample_t)(volts / 5.0f / 3.3f * (float)ADC_HTR_HT + 0.5f);
90 }
