68 /**
69  * @brief Driver possible handling of counter overflow/underflow.
70  *
71  * @details When counter is going to overflow, the new value is
72  * computed according to this mode in such a way that
73  * the counter will either wrap around, stay unchange
74  * or reach min/max
75  *
76  * @note All driver implementation should support the
77  * QEI_OVERFLOW_WRAP mode.
78  *
79  * @note Mode QEI_OVERFLOW_DISCARD and QEI_OVERFLOW_MINMAX are included
80  * if QEI_USE_OVERFLOW_DISCARD and QEI_USE_OVERFLOW_MINMAX are
81  * set to TRUE in halconf_community.h and are not necessary supported
82  * by all drivers
83  */
84 typedef enum {
85  QEI_OVERFLOW_WRAP = 0, /**< Counter value will wrap around. */
86 #if defined(QEI_USE_OVERFLOW_DISCARD) && QEI_USE_OVERFLOW_DISCARD == TRUE
87  QEI_OVERFLOW_DISCARD = 1, /**< Counter doesn't change. */
88 #endif
89 #if defined(QEI_USE_OVERFLOW_MINMAX) && QEI_USE_OVERFLOW_MINMAX == TRUE
90  QEI_OVERFLOW_MINMAX = 2, /**< Counter will be updated upto min or max.*/
91 #endif
92 } qeioverflow_t;
