45 /**
46  * @brief Helper for correclty handling overflow/underflow
47  *
48  * @details Underflow/overflow will be handled according to mode:
49  * QEI_OVERFLOW_WRAP: counter value will wrap around.
50  * QEI_OVERFLOW_DISCARD: counter will not change
51  * QEI_OVERFLOW_MINMAX: counter will be updated upto min or max.
52  *
53  * @note This function is for use by low level driver.
54  *
55  * @param[in,out] count counter value
56  * @param[in,out] delta adjustment value
57  * @param[in] min minimum allowed value for counter
58  * @param[in] max maximum allowed value for counter
59  * @param[in] mode how to handle overflow
60  *
61  * @return true if counter underflow/overflow occured or
62  * was due to occur
63  *
64  */
65 static inline
66 bool qei_adjust_count(qeicnt_t *count, qeidelta_t *delta,
67  qeicnt_t min, qeicnt_t max, qeioverflow_t mode) {
68  /* For information on signed integer overflow see:
69  * https://www.securecoding.cert.org/confluence/x/RgE
70  */
71 
72  /* Get values */
73  const qeicnt_t _count = *count;
74  const qeidelta_t _delta = *delta;
75 
76  /* Overflow operation
77  */
78  if ((_delta > 0) && (_count > (max - _delta))) {
79  switch(mode) {
80  case QEI_OVERFLOW_WRAP:
81  *delta = 0;
82  *count = (min + (_count - (max - _delta))) - 1;
83  break;
84 #if QEI_USE_OVERFLOW_DISCARD == TRUE
85  case QEI_OVERFLOW_DISCARD:
86  *delta = _delta;
87  *count = _count;
88  break;
89 #endif
90 #if QEI_USE_OVERFLOW_MINMAX == TRUE
91  case QEI_OVERFLOW_MINMAX:
92  *delta = _count - (max - _delta);
93  *count = max;
94  break;
95 #endif
96  }
97  return true;
98 
99  /* Underflow operation
100  */
101  } else if ((_delta < 0) && (_count < (min - _delta))) {
102  switch(mode) {
103  case QEI_OVERFLOW_WRAP:
104  *delta = 0;
105  *count = (max + (_count - (min - _delta))) + 1;
106  break;
107 #if QEI_USE_OVERFLOW_DISCARD == TRUE
108  case QEI_OVERFLOW_DISCARD:
109  *delta = _delta;
110  *count = _count;
111  break;
112 #endif
113 #if QEI_USE_OVERFLOW_MINMAX == TRUE
114  case QEI_OVERFLOW_MINMAX:
115  *delta = _count - (min - _delta);
116  *count = min;
117  break;
118 #endif
119  }
120  return true;
121 
122  /* Normal operation
123  */
124  } else {
125  *delta = 0;
126  *count = _count + _delta;
127  return false;
128  }
129 }

294 /**
295  * @brief Adjust the counter by delta.
296  *
297  * @param[in] qeip pointer to the @p QEIDriver object.
298  * @param[in] delta the adjustement value.
299  * @return the remaining delta (can occur during overflow).
300  *
301  * @api
302  */
303 qeidelta_t qeiAdjustI(QEIDriver *qeip, qeidelta_t delta) {
304  /* Get boundaries */
305  qeicnt_t min = QEI_COUNT_MIN;
306  qeicnt_t max = QEI_COUNT_MAX;
307  if (qeip->config->min != qeip->config->max) {
308  min = qeip->config->min;
309  max = qeip->config->max;
310  }
311 
312  /* Get counter */
313  qeicnt_t count = qei_lld_get_count(qeip);
314 
315  /* Adjust counter value */
316  bool overflowed = qei_adjust_count(&count, &delta,
317  min, max, qeip->config->overflow);
318 
319  /* Notify for value change */
320  qei_lld_set_count(qeip, count);
321 
322  /* Notify for overflow (passing the remaining delta) */
323  if (overflowed && qeip->config->overflow_cb)
324  qeip->config->overflow_cb(qeip, delta);
325 
326  /* Remaining delta */
327  return delta;
328 }

349 /**

355  * @iclass
356  */
357 qeidelta_t qeiUpdateI(QEIDriver *qeip) {
358  qeicnt_t cnt;
359  qeidelta_t delta;
360 
361  osalDbgCheckClassI();
362  osalDbgCheck(qeip != NULL);
363  osalDbgAssert((qeip->state == QEI_READY) || (qeip->state == QEI_ACTIVE),
364  "invalid state");
365 
366  cnt = qei_lld_get_count(qeip);
367  delta = (qeicnt_t)(cnt - qeip->last);
368  qeip->last = cnt;
369 
370  return delta;
